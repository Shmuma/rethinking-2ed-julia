@model function m14_1(cafe, afternoon, wait)
    a ~ Normal(5, 2)
    b ~ Normal(-1, 0.5)
    _cafe ~ filldist(Exponential(), 2)
    Rho ~ LKJ(2, 2)
    # build sigma matrix manually, to avoid numerical errors
    (, ) = _cafe
    sc = [[^2, *] [*, ^2]]
     = Rho .* sc
    ab ~ filldist(MvNormal([a,b], ), N_cafes)
    a = ab[1,cafe]
    b = ab[2,cafe]
     = @. a + b * afternoon
     ~ Exponential()
    for i  eachindex(wait)
        wait[i] ~ Normal([i], )
    end
end
