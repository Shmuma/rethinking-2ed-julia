@model function m11_2(pulled_left, treatment)
    a ~ Normal(0, 1.5)
    treat_levels = length(levels(d.treatment))
    b ~ MvNormal(zeros(treat_levels), 10)
    
    p = @. logistic(a + b[treatment])
    for i ∈ eachindex(pulled_left)
        pulled_left[i] ~ Binomial(1, p[i])
    end
end
